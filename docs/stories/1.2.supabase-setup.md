    ```typescript
    import { NextResponse, type NextRequest } from 'next/server'

    export async function updateSession(request: NextRequest) {
      let supabaseResponse = NextResponse.next({ request })

      const supabase = createServerClient(
        process.env.NEXT_PUBLIC_SUPABASE_URL!,
        process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
        {
          cookies: {
            getAll() {
              return request.cookies.getAll()
            },
            setAll(cookiesToSet) {
              cookiesToSet.forEach(({ name, value }) =>
                request.cookies.set(name, value)
              )
              supabaseResponse = NextResponse.next({ request })
              cookiesToSet.forEach(({ name, value, options }) =>
                supabaseResponse.cookies.set(name, value, options)
              )
            },
          },
        }
      )

      await supabase.auth.getUser()

      return supabaseResponse
    }
    ```

### Supabase Client Pattern (Next.js App Router)

The Supabase client uses `@supabase/ssr` for cookie-based auth in Next.js App Router. Three clients are needed:

| Client | File | Usage | Auth |
|--------|------|-------|------|
| **Browser** | `client.ts` | Client Components (`'use client'`) | User session via cookies |
| **Server** | `server.ts` | Server Components, Route Handlers | User session via cookies |
| **Admin** | `admin.ts` | Webhooks only (Pluggy, Stripe) | Service role (bypasses RLS) |

### Important Security Rules

- **Browser client** uses ONLY `NEXT_PUBLIC_SUPABASE_URL` and `NEXT_PUBLIC_SUPABASE_ANON_KEY`
- **Service role key** (`SUPABASE_SERVICE_ROLE_KEY`) MUST NEVER be in browser client — it bypasses ALL RLS
- **Middleware** refreshes auth session on every request — required for SSR auth to work
- **Cookie-based auth** is the correct pattern for Next.js App Router (NOT localStorage)
